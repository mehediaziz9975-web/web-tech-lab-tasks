  <script>
    function addBook() {
      const title = document.getElementById('title').value.trim();
      const year = document.getElementById('year').value.trim();
      const currentYear = new Date().getFullYear();

      let isValid = true;

      // Reset errors
      document.getElementById('titleError').innerText = "";
      document.getElementById('yearError').innerText = "";

      // Validate title
      if (title === "" || !/^[A-Za-z]+$/.test(title)) {
        document.getElementById('titleError').innerText = "Title must contain only alphabets and no spaces.";
        isValid = false;
      }

      // Validate year
      if (!/^\d{4}$/.test(year) || year < 1900 || year > currentYear) {
        document.getElementById('yearError').innerText = "Year must be between 1900 and " + currentYear + ".";
        isValid = false;
      }

      // If valid, add to table
      if (isValid) {
        const table = document.getElementById('bookTable');
        const row = document.createElement('tr');

        if (year < 2000) {
          row.classList.add('old-book');
        } else {
          row.classList.add('new-book');
        }

        row.innerHTML = `<td>${title}</td><td>${year}</td>`;
        table.appendChild(row);

        // Clear inputs
        document.getElementById('title').value = "";
        document.getElementById('year').value = "";
      }
    }
  </script>
